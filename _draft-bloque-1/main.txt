  // Smooth scrolling function
  function smoothScroll(target) {
    const element = document.querySelector(target);
    if (element) {
      element.scrollIntoView({
        behavior: 'smooth',
        block: 'start'
      });
    }
  }

  // Function to show only the specified content section and hide others
  function showOnlyContent(contentId) {
    const contents = document.querySelectorAll('.content');
    contents.forEach(content => {
      if (content.id === contentId) {
        content.style.display = 'block';
      } else {
        content.style.display = 'none';
      }
    });
  }

  // Event listener for menu items
  document.addEventListener('click', function (event) {
    if (event.target.classList.contains('menu-item')) {
      event.preventDefault();
      const target = event.target.getAttribute('data-target');
      smoothScroll(target);

      // Show the corresponding content sections and hide others
      showOnlyContent(target.substring(1));
      setActiveMenuItem(target); // Update the active menu item based on the data-target
    }
  });

  // Function to add "active-menu-item" class to the selected menu item
  function setActiveMenuItem(target) {
    const items = document.querySelectorAll('.menu-item');
    items.forEach(item => {
      if (item.getAttribute('data-target') === target) {
        item.classList.add('active-menu-item');
      } else {
        item.classList.remove('active-menu-item');
      }
    });
  }

  // Function to show the specified content section and update the active menu item
  function showContent(contentId) {
    const contents = document.querySelectorAll('.content');
    contents.forEach(content => {
      if (content.id === contentId) {
        content.style.display = 'block';
        setActiveMenuItem('#' + contentId); // Set the active menu item based on the contentId
      } else {
        content.style.display = 'none';
      }
    });
  }

  // Add functions to select the previous or next menu item
  function selectPreviousItem() {
    const items = document.querySelectorAll('.menu-item');
    let currentIndex = -1;

    // Find the index of the currently active menu item
    for (let i = 0; i < items.length; i++) {
      if (items[i].classList.contains('active-menu-item')) {
        currentIndex = i;
        break;
      }
    }

    if (currentIndex !== -1) {
      // Remove 'active-menu-item' class from the current menu item
      items[currentIndex].classList.remove('active-menu-item');

      // Calculate the index of the previous menu item (or loop to the last item if the first item is active)
      const previousIndex = currentIndex === 0 ? items.length - 1 : currentIndex - 1;

      // Add 'active-menu-item' class to the previous menu item
      items[previousIndex].classList.add('active-menu-item');

      // Display the corresponding content section
      showContent(items[previousIndex].getAttribute('data-target').substring(1));
    }
  }

  function selectNextItem() {
    const items = document.querySelectorAll('.menu-item');
    let currentIndex = -1;

    // Find the index of the currently active menu item
    for (let i = 0; i < items.length; i++) {
      if (items[i].classList.contains('active-menu-item')) {
        currentIndex = i;
        break;
      }
    }

    if (currentIndex !== -1) {
      // Remove 'active-menu-item' class from the current menu item
      items[currentIndex].classList.remove('active-menu-item');

      // Calculate the index of the next menu item (or loop to the first item if the last item is active)
      const nextIndex = currentIndex === items.length - 1 ? 0 : currentIndex + 1;

      // Add 'active-menu-item' class to the next menu item
      items[nextIndex].classList.add('active-menu-item');

      // Display the corresponding content section
      showContent(items[nextIndex].getAttribute('data-target').substring(1));
    }
  }

  // Show the content for menu item 1 by default
  showContent('content-block1');
